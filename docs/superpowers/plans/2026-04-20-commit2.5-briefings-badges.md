```tsx
"use client";

import { forwardRef, useEffect, useRef } from "react";
import { cn } from "@/lib/utils";

type AutoTextareaProps = Omit<
  React.TextareaHTMLAttributes<HTMLTextAreaElement>,
  "ref"
> & {
  /** Altura mínima em px. Default 60. */
  minHeight?: number;
};

/**
 * Textarea que cresce com o conteúdo. Usa `scrollHeight` pra recalcular
 * a altura em cada render/input. `resize: none` no CSS porque o controle
 * é automático.
 */
export const AutoTextarea = forwardRef<HTMLTextAreaElement, AutoTextareaProps>(
  function AutoTextarea({ minHeight = 60, className, value, onChange, ...rest }, ref) {
    const innerRef = useRef<HTMLTextAreaElement | null>(null);

    function setRefs(el: HTMLTextAreaElement | null) {
      innerRef.current = el;
      if (typeof ref === "function") ref(el);
      else if (ref) (ref as React.MutableRefObject<HTMLTextAreaElement | null>).current = el;
    }

    function resize() {
      const el = innerRef.current;
      if (!el) return;
      el.style.height = "0px";
      el.style.height = `${Math.max(minHeight, el.scrollHeight)}px`;
    }

    // Resize ao montar e sempre que `value` mudar (controlled)
    useEffect(() => {
      resize();
    }, [value, minHeight]);

    return (
      <textarea
        ref={setRefs}
        value={value}
        onChange={(e) => {
          onChange?.(e);
          // resize imediato pra sensação instantânea (sem esperar o useEffect)
          requestAnimationFrame(resize);
        }}
        style={{ minHeight, resize: "none" }}
        className={cn(
          "w-full overflow-hidden rounded-md border border-border bg-background px-2 py-1.5 text-xs outline-none focus:ring-2 focus:ring-mint-500/30",
          className,
        )}
        {...rest}
      />
    );
  },
);
```

"use client";

import { useState, useTransition } from "react";
import { X, Paperclip, Link as LinkIcon, Check, Trash2 } from "lucide-react";
import { cn } from "@/lib/utils";
import type { Briefing, BriefingAttachment } from "@/lib/types";
import { AutoTextarea } from "./auto-textarea";
import {
  upsertBriefingAction,
  deleteBriefingAction,
  setBriefingStatusAction,
} from "@/lib/campaigns/rich-actions";

/**
 * Modal de detalhe de um briefing (seja edit de existente ou criação de novo).
 *
 * Fluxo:
 * - `briefing` vem completo do parent; user edita state local.
 * - "Salvar" chama upsertBriefingAction.
 * - "Feito" alterna status pending<->done (action dedicada).
 * - "Anexar" pede URL via prompt simples (sem upload real — ver nota abaixo).
 *
 * Nota sobre anexos: a UI do mockup mostra ícones de clip e link; por agora
 * usamos URL input (prompt) pra ambos — upload real pra storage fica pra
 * iteração futura. Isso mantém o Commit 2.5 focado no card/briefing shape.
 */
export function BriefingDetailModal({
  stepId,
  briefing,
  onClose,
  onSaved,
  onDeleted,
}: {
  stepId: string;
  briefing: Briefing;
  onClose: () => void;
  onSaved: (next: Briefing) => void;
  onDeleted: (id: string) => void;
}) {
  const [label, setLabel] = useState(briefing.label);
  const [date, setDate] = useState<string>(briefing.date ?? "");
  const [planning, setPlanning] = useState(briefing.planning);
  const [publishing, setPublishing] = useState(briefing.publishing);
  const [status, setStatus] = useState(briefing.status);
  const [attachments, setAttachments] = useState<BriefingAttachment[]>(
    briefing.attachments,
  );
  const [pending, start] = useTransition();
  const [error, setError] = useState<string | null>(null);

  function save() {
    setError(null);
    const next: Briefing = {
      id: briefing.id,
      label: label.trim() || `Briefing`,
      date: date || null,
      planning,
      publishing,
      status,
      attachments,
    };
    start(async () => {
      const res = await upsertBriefingAction({
        step_id: stepId,
        id: next.id,
        label: next.label,
        date: next.date ?? undefined,
        planning: next.planning,
        publishing: next.publishing,
        status: next.status,
        attachments: next.attachments,
      });
      if (res?.error) setError(res.error);
      else {
        onSaved(next);
        onClose();
      }
    });
  }

  function toggleDone() {
    const nextStatus = status === "done" ? "pending" : "done";
    setStatus(nextStatus);
    start(async () => {
      await setBriefingStatusAction({
        step_id: stepId,
        briefing_id: briefing.id,
        status: nextStatus,
      });
    });
  }

  function addAttachment(kind: "file" | "link") {
    const promptLabel = kind === "file" ? "URL do arquivo" : "URL do link";
    const url = window.prompt(promptLabel);
    if (!url) return;
    const labelText = window.prompt("Rótulo (opcional)") ?? "";
    setAttachments((a) => [
      ...a,
      { url, label: labelText || (kind === "file" ? "Arquivo" : "Link") },
    ]);
  }

  function removeAttachment(url: string) {
    setAttachments((a) => a.filter((x) => x.url !== url));
  }

  function del() {
    if (!window.confirm("Remover este briefing?")) return;
    start(async () => {
      const res = await deleteBriefingAction(stepId, briefing.id);
      if (res?.error) setError(res.error);
      else {
        onDeleted(briefing.id);
        onClose();
      }
    });
  }

  return (
    <div className="fixed inset-0 z-50 grid place-items-center bg-black/40 p-4 backdrop-blur-sm">
      <div className="relative w-full max-w-xl rounded-2xl border border-border bg-card p-6 shadow-xl">
        <button
          type="button"
          onClick={onClose}
          className="absolute right-4 top-4 rounded p-1 text-muted-foreground hover:bg-muted"
          aria-label="Fechar"
        >
          <X className="h-4 w-4" />
        </button>

        <div className="flex items-center justify-between gap-3 pr-8">
          <input
            value={label}
            onChange={(e) => setLabel(e.target.value)}
            maxLength={40}
            className="min-w-0 flex-1 border-none bg-transparent text-lg font-semibold tracking-tight outline-none"
            placeholder="Título do briefing"
          />
          <span
            className={cn(
              "inline-flex items-center rounded-full px-2 py-0.5 text-[11px] font-semibold",
              status === "done"
                ? "bg-mint-100 text-mint-900 dark:bg-mint-900/40 dark:text-mint-200"
                : "bg-red-100 text-red-900 dark:bg-red-900/40 dark:text-red-200",
            )}
          >
            {status === "done" ? "Feito" : "Pendente"}
          </span>
        </div>

        <label className="mt-4 block space-y-1">
          <span className="text-[11px] font-medium uppercase tracking-wide text-muted-foreground">
            Data
          </span>
          <input
            type="date"
            value={date}
            onChange={(e) => setDate(e.target.value)}
            className="w-full rounded-lg border border-border bg-background px-3 py-2 text-sm outline-none focus:ring-2 focus:ring-mint-500/30"
          />
        </label>

        <Section
          title="Planejamento"
          value={planning}
          onChange={setPlanning}
          onAttach={() => addAttachment("file")}
          onLink={() => addAttachment("link")}
        />
        <Section
          title="Publicação"
          value={publishing}
          onChange={setPublishing}
          onAttach={() => addAttachment("file")}
          onLink={() => addAttachment("link")}
        />

        {attachments.length > 0 && (
          <div className="mt-3 space-y-1">
            <span className="text-[11px] font-medium uppercase tracking-wide text-muted-foreground">
              Anexos
            </span>

"use client";

import { useState, useTransition } from "react";
import { ChevronDown, Calendar } from "lucide-react";
import { cn } from "@/lib/utils";
import type { Step, Briefing, StepStatusV2 } from "@/lib/types";
import { TrophyToggle } from "./trophy-toggle";
import { BriefingBadges } from "./briefing-badges";
import { BriefingDetailModal } from "./briefing-detail-modal";
import { setStepStatusV2Action } from "@/lib/campaigns/rich-actions";

/**
 * TaskCardRich — card com BriefingBadges numeradas.
 *
 * Não renderiza seu próprio frame — width/height vêm do parent (react-rnd).
 * Usa sticky header + overflow-y-auto pra manter título visível ao scroll.
 */
export function TaskCardRich({ step }: { step: Step }) {
  const [briefings, setBriefings] = useState<Briefing[]>(step.briefings ?? []);
  const [status, setStatus] = useState<StepStatusV2>(step.status_v2);
  const [menuOpen, setMenuOpen] = useState(false);
  const [openBriefingId, setOpenBriefingId] = useState<string | null>(null);
  const [pending, start] = useTransition();

  const openBriefing =
    openBriefingId === null ? null : briefings.find((b) => b.id === openBriefingId) ?? null;

  function addBriefing() {
    const id = crypto.randomUUID();
    const blank: Briefing = {
      id,
      label: `Briefing ${briefings.length + 1}`,
      date: null,
      planning: "",
      publishing: "",
      status: "pending",
      attachments: [],
    };
    setBriefings((b) => [...b, blank]);
    setOpenBriefingId(id); // abre direto o detail modal
  }

  function onBriefingSaved(next: Briefing) {
    setBriefings((bs) => {
      const idx = bs.findIndex((b) => b.id === next.id);
      if (idx === -1) return [...bs, next];
      const copy = bs.slice();
      copy[idx] = next;
      return copy;
    });
  }

  function onBriefingDeleted(id: string) {
    setBriefings((bs) => bs.filter((b) => b.id !== id));
  }

  function setStatusUi(s: StepStatusV2) {
    setStatus(s);
    setMenuOpen(false);
    start(async () => {
      await setStepStatusV2Action({ step_id: step.id, status: s });
    });
  }

  return (
    <>
      <div
        className={cn(
          "flex h-full flex-col overflow-hidden rounded-2xl border bg-card shadow-sm transition",
          status === "done" && "border-mint-500",
          status === "publishing" && "border-amber-400",
          status === "planning" && "border-border",
        )}
      >
        <header className="sticky top-0 z-10 flex items-center justify-between gap-2 border-b border-border bg-card/95 px-3 py-2 backdrop-blur">
          <h3 className="truncate text-sm font-semibold" title={step.title}>
            {step.title}
          </h3>
          <div className="flex shrink-0 items-center gap-1.5">
            <TrophyToggle
              stepId={step.id}
              valueUsd={step.trophy_value_usd}
              multiplierX={step.trophy_multiplier_x}
            />
            <div className="relative">
              <button
                type="button"
                onClick={() => setMenuOpen((o) => !o)}
                disabled={pending}
                className={cn(
                  "inline-flex items-center gap-1 rounded-full px-2 py-0.5 text-[11px] font-medium",
                  status === "done" &&
                    "bg-mint-100 text-mint-900 dark:bg-mint-900/50 dark:text-mint-200",
                  status === "publishing" &&
                    "bg-amber-100 text-amber-900 dark:bg-amber-900/40 dark:text-amber-200",
                  status === "planning" && "bg-muted text-muted-foreground",
                )}
              >
                {statusLabel(status)} <ChevronDown className="h-3 w-3" />
              </button>
              {menuOpen && (
                <div className="absolute right-0 top-full z-20 mt-1 w-36 rounded-lg border border-border bg-card p-1 shadow-lg">
                  {(["planning", "publishing", "done"] as const).map((s) => (
                    <button
                      key={s}
                      type="button"
                      onClick={() => setStatusUi(s)}
                      className={cn(
                        "flex w-full items-center gap-2 rounded-md px-2 py-1 text-left text-xs hover:bg-muted",
                        s === status && "bg-muted font-medium",
                      )}
                    >
                      {statusLabel(s)}
                    </button>
                  ))}
                </div>
              )}
            </div>
          </div>
        </header>

        <div className="flex-1 space-y-3 overflow-y-auto p-3">
          {step.description && (
            <p className="text-xs text-muted-foreground">{step.description}</p>
          )}

          {step.due_date && (
            <div className="inline-flex items-center gap-1 rounded-full bg-muted px-2 py-0.5 text-[11px] text-muted-foreground">
              <Calendar className="h-3 w-3" />
              {new Date(step.due_date).toLocaleDateString("pt-BR", {
                day: "2-digit",
                month: "short",
              })}
            </div>
          )}

          <BriefingBadges
            briefings={briefings}
            onOpen={(id) => setOpenBriefingId(id)}
            onAdd={addBriefing}
          />
        </div>
      </div>

      {openBriefing && (
        <BriefingDetailModal
          stepId={step.id}
          briefing={openBriefing}
          onClose={() => setOpenBriefingId(null)}
          onSaved={onBriefingSaved}
          onDeleted={onBriefingDeleted}
        />
      )}
    </>
  );
}

function statusLabel(s: StepStatusV2): string {
  return s === "done" ? "Feito" : s === "publishing" ? "Publicação" : "Planejamento";
}

"use client";

import { useState, useTransition } from "react";

import type { CampaignInfo } from "@/lib/types";
import { upsertCampaignInfoAction } from "@/lib/campaigns/info-actions";
import { AutoTextarea } from "./auto-textarea";

const LOGIN_OPTIONS = ["Gmail", "X", "Wallet", "Outro"] as const;
type LoginOption = (typeof LOGIN_OPTIONS)[number];

function normalizeLogin(raw: string | null): LoginOption {
  if (!raw) return "Gmail";
  return (LOGIN_OPTIONS as readonly string[]).includes(raw)
    ? (raw as LoginOption)
    : "Outro";
}

export function InfoPanel({
  campaignId,
  initial,
}: {
  campaignId: string;
  initial: CampaignInfo | null;
}) {
  const [login, setLogin] = useState<LoginOption>(normalizeLogin(initial?.login ?? null));
  const [blockchain, setBlockchain] = useState(initial?.blockchain_ref ?? "");
  const [wallet, setWallet] = useState(initial?.wallet ?? "");
  const [pending, start] = useTransition();
  const [saved, setSaved] = useState(false);

  function save() {
    const fd = new FormData();
    fd.set("campaign_id", campaignId);
    fd.set("login", login);
    fd.set("blockchain_ref", blockchain);
    fd.set("wallet", wallet);
    start(async () => {
      await upsertCampaignInfoAction(fd);
      setSaved(true);
      setTimeout(() => setSaved(false), 1500);
    });
  }

  return (
    <aside className="rounded-2xl border border-border bg-card p-4">
      <header className="mb-3 flex items-center justify-between">
        <h3 className="text-sm font-semibold">Info</h3>
        <button

          onClick={save}
          disabled={pending}
          className="inline-flex items-center gap-1 rounded-md bg-mint-600 px-2 py-1 text-[11px] font-medium text-white hover:bg-mint-700 disabled:opacity-60"

          <Save className="h-3 w-3" />{" "}
          {saved ? "Salvo" : pending ? "Salvando" : "Salvar"}
        </button>
      </header>

      <label className="mb-3 block">
        <span className="mb-1 inline-flex items-center gap-1.5 text-[11px] font-medium uppercase tracking-wide text-muted-foreground">
          <KeyRound className="h-3 w-3" /> Login usado

        <select
          value={login}
          onChange={(e) => setLogin(e.target.value as LoginOption)}
          className="w-full rounded-md border border-border bg-background px-2 py-1.5 text-xs outline-none focus:ring-2 focus:ring-mint-500/30"
        >
          {LOGIN_OPTIONS.map((o) => (
